    &str е’Ң String

    &str

    • &str жҳҜеӯ—з¬ҰдёІеҲҮзүҮпјҢжҳҜеҲҮзүҮзҡ„дёҖз§ҚгҖӮ
    • еҪўеҰӮ "string literals" зҡ„еӯ—з¬ҰдёІеӯ—йқўеҖјжҳҜ &str зұ»еһӢзҡ„1гҖӮ
    • &str жҳҜйқҷжҖҒеҲҶй…Қз©әй—ҙзҡ„пјҢдё”еӣәе®ҡеӨ§е°ҸгҖӮ
    • дёҚиғҪз”Ёж–№жӢ¬еҸ·жқҘеҒҡеҪўеҰӮ some_str[i] зҡ„зҙўеј•пјҢеӣ дёәжҜҸдёӘ Unicode еӯ—з¬ҰеҸҜиғҪжңүеӨҡдёӘеӯ—иҠӮгҖӮ
    • жӯЈзЎ®зҡ„еҒҡжі•жҳҜеңЁ chars() дёӯиҝӯд»Јпјҡ
    • for c in "1234".chars() { ... }

    String

    • String жҳҜеҲҶй…ҚеңЁе ҶдёҠзҡ„пјҢеҸҜд»ҘеҠЁжҖҒеўһй•ҝгҖӮ
    • е’Ң Vec зұ»дјјпјҢе®һйҷ…дёҠе°ұжҳҜеңЁ Vec\<u8> еӨ–йқўеҢ…дәҶдёҖеұӮгҖӮ
    • д№ҹдёҚиғҪз”ЁдёӢж ҮжқҘзҙўеј•гҖӮ
    • еҸҜд»ҘйҖҡиҝҮ s.nth(i) жқҘи®ҝй—®жҹҗдёӘеӯ—з¬ҰгҖӮ
    • йҖҡиҝҮеҸ–еј•з”Ёзҡ„ж–№ејҸеҸҜд»ҘиҺ·еҫ— &strгҖӮ

    Option жһҡдёҫзұ»еһӢ

    enum Option<T> {
  None,
  Some(T),
}
    • Option жҳҜдёҖдёӘжһҡдёҫзұ»еһӢпјҢеҗҢж—¶д№ҹжҳҜжіӣеһӢзұ»еһӢгҖӮ
    • дёәжҹҗз§Қе·Іжңүзұ»еһӢжҸҗдҫӣдәҶиЎЁзӨәжІЎжңүжҲ–иҖ…з©әеҖјзҡ„жҰӮеҝөгҖӮ
    • еңЁ Rust дёӯпјҢеңЁйңҖиҰҒиҝ”еӣһз©әеҖјж—¶пјҢжҺЁиҚҗдҪҝз”Ё OptionгҖӮ
    • иҖҢдёҚжҳҜиҝ”еӣһиҜёеҰӮ NaNгҖҒ -1гҖҒ null зӯүзү№ж®Ҡзҡ„еҖјгҖӮ
    • зұ»еһӢ T еҸҜд»ҘжҳҜд»»дҪ•зұ»еһӢпјҢжІЎжңүйҷҗеҲ¶гҖӮ

    дёҖдёӘеӨ„зҗҶйҷӨж•°дёә 0 зҡ„жғ…еҶөпјҡ

    fn divide(numerator: f64, denominator: f64) -> Option<f64> {
if denominator == 0.0 {
    None
  } else {
    Some(numerator / denominator)
  }
}
// The return value of the function is an option
let result = divide(2.0, 3.0);
// Pattern match to retrieve the value
match result {
  // The division was valid
  Some(x) => println!("Result: {x}"),
  // The division was invalid
  None => println!("Cannot divide by 0"),
}
